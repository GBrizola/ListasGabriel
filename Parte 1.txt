//Definindo vari√°veis
int bits = 0;
int led = 3;
int tempo = 50;

//void Setup
void setup(){
	pinMode(led, OUTPUT);
  
  //Monitor Serial
  Serial.begin(9600);
}

//void loop
void loop(){
  for(int i = 0; i < 256; i++){
    analogWrite(led,i);
    delay(tempo);
  }
  for(int j = 256; j > 0; j--){
    analogWrite(led,j);
    delay(tempo);
  }
}
